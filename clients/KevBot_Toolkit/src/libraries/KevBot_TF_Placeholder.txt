// Returns placeholder values for testing the toolkit wiring
//━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

export getTFConfluence(
     string tf1_res, string tf2_res, string tf3_res,
     string tf4_res, string tf5_res, string tf6_res,
     float  paramA,  float paramB,  float paramC,
     float  paramD,  float paramE,  float paramF) =>

    // Use params to keep them alive
    float _sum = paramA + paramB + paramC + paramD + paramE + paramF
    bool _useParams = _sum == _sum

    // Placeholder conditions based on paramA threshold
    bool cA = paramA > 1
    bool cB = paramA > 2
    bool cC = paramA > 3
    bool cD = paramA > 4
    bool cE = paramA > 5
    bool cF = paramA > 6

    // Placeholder triggers
    bool tA = paramA > 0
    bool tB = paramA > 10

    TFModuleOutput.new(
        // Labels
        "TF1-" + tf1_res, "TF2-" + tf2_res, "TF3-" + tf3_res,
        "TF4-" + tf4_res, "TF5-" + tf5_res, "TF6-" + tf6_res,
        // Cond A
        cA, cA, cA, cA, cA, cA,
        // Cond B
        cB, cB, cB, cB, cB, cB,
        // Cond C
        cC, cC, cC, cC, cC, cC,
        // Cond D
        cD, cD, cD, cD, cD, cD,
        // Cond E
        cE, cE, cE, cE, cE, cE,
        // Cond F
        cF, cF, cF, cF, cF, cF,
        // Cond G-J (all false)
        false, false, false, false, false, false,
        false, false, false, false, false, false,
        false, false, false, false, false, false,
        false, false, false, false, false, false,
        // Triggers A-J
        tA, tB, false, false, false, false, false, false, false, false,
        // Trigger metadata
        tA ? close : na,
        "Placeholder (" + str.tostring(_sum) + ")"
    )
